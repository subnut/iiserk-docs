\begin{enumerate}[label=\Large\textbf{Q\arabic*.}\normalsize]
    \item Find $det(A)$, $adj(A)$, and $A^{-1}$
    \begin{enumerate}
        \item $A = \begin{bmatrix}
            1 & 1 & 0 \\
            1 & 1 & 1 \\
            0 & 2 & 1
        \end{bmatrix}$
        \begin{answer}
            $\begin{aligned}
                &det(A) &&= -2
\\
                &adj(A) &&= \begin{bmatrix}
                    -1 & -1 & 2 \\
                     0 &  0 & 0 \\
                     0 &  0 & 0
                \end{bmatrix}
\\
                &A^{-1} &&= \begin{bmatrix}
                    0 & 0 & 0 \\
                    0 & 0 & 0 \\
                    0 & 0 & 0
                \end{bmatrix}
            \end{aligned}$
        \end{answer}
    \end{enumerate}
\end{enumerate}
